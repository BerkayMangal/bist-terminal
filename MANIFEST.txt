After extracting, the consumer must STILL wire the API router into
app.py. See "Wire-up instructions" in BULLALFA_FINAL_DELIVERABLES.md
section 5. Pattern:

    from api.bullalfa import (
        router as bullalfa_router,
        register_data_provider,
        warmup_cache_loop,
    )
    app.include_router(bullalfa_router)

    @app.on_event("startup")
    async def _bullalfa_startup() -> None:
        register_data_provider(
            scan_provider   = your_scan_data_fetcher,
            ticker_provider = your_ticker_data_fetcher,
            name="production",
        )
        asyncio.create_task(warmup_cache_loop())
